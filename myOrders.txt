            measurementId: "G-QX5HCKXMDK"
        };

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);

        const queryParams = new URLSearchParams(window.location.search);
        const mobile = queryParams.get('mobile');

        async function loadOrders() {
            const ordersRef = collection(db, 'users', mobile, 'orders');
            const querySnapshot = await getDocs(ordersRef);
            const ordersContainer = document.getElementById('ordersContainer');
            ordersContainer.innerHTML = '';

            const sortedDocs = querySnapshot.docs.sort((a, b) => {
                return a.data().timestamp.toDate() - b.data().timestamp.toDate();
            });

            sortedDocs.forEach(doc => {
                const order = doc.data();
                const orderElement = document.createElement('div');
                orderElement.className = 'order-item';

                // Get and format the timestamp
                const orderDate = order.timestamp.toDate();
                const formattedTime = orderDate.toLocaleString();

                // Create a container for the timestamp
                const timestampElement = document.createElement('div');
                timestampElement.className = 'order-timestamp';
                timestampElement.innerHTML = `<strong>Date:</strong><pre> </pre> ${formattedTime}<br>`;

                // Append the timestamp to the order element
                orderElement.appendChild(timestampElement);

                order.order.forEach(item => {
                    // Shorten the item name if it exceeds a certain length
                    const maxLength = 15; // Define the max number of characters
                    let itemName = item.itemName;

                    // Function to truncate text at word or punctuation boundary
                    function truncateText(text, maxLength) {
                        if (text.length <= maxLength) return text;
                        let truncated = text.slice(0, maxLength);
                        const lastChar = truncated.slice(-1);
                        if (lastChar !== ' ' && lastChar !== '-' && lastChar !== '.') {
                            // Ensure truncation happens at a space or hyphen if possible
                            const lastSpaceIndex = truncated.lastIndexOf(' ');
                            const lastHyphenIndex = truncated.lastIndexOf('-');
                            const truncationIndex = Math.max(lastSpaceIndex, lastHyphenIndex);
                            truncated = text.slice(0, truncationIndex > 0 ? truncationIndex : maxLength);
                        }
                        return truncated + ' ..';
                    }

                    itemName = truncateText(itemName, maxLength);

                    const itemElement = document.createElement('div');
                    itemElement.innerHTML = `
            <div style='flex-direction:row'>
                <div class='kushu'>
                    <img src="${item.itemImage}" alt="${itemName}">
                    <div class="nameQuantityPrice">
                        <div class='itemname'><span><strong>${itemName}</strong></span></div>
                        <div class='itemquantity'><span>${item.quantity}</span></div>
                        <div class='itemprice'><span><div class='itemPriceInsideText2'>Rs. ${item.itemPrice}</span></div></div>
                    </div>
                </div>
            </div>
            `;
                    orderElement.appendChild(itemElement);
                });

                orderElement.innerHTML += `
        <div class="status">
            <div class='orderStatusText'>
                <div class='orderStatusInsideText'><strong>Status:</strong></div> 
                <div class='orderStatusInsideText2'>${order.status}<br></div>
            </div>
            <div class='orderTotalAmountText'>
                <div class='orderTotalAmountInsideText'><strong>Total Amount:</strong></div> 
                <div class='orderTotalAmountInsideText2'>Rs. ${order.totalAmount}</div>
            </div>
        </div>`;

                ordersContainer.appendChild(orderElement);
                const statusValueElement = orderElement.querySelector('.orderStatusInsideText2');
                if (statusValueElement.textContent.includes('Approved')) {
                    statusValueElement.style.color = 'lightgreen';
                    orderElement.style.borderLeftColor = 'lightgreen';
                } else if (statusValueElement.textContent.includes('Pending')) {
                    statusValueElement.style.color = 'rgb(255, 255, 13)';
                    orderElement.style.borderLeftColor = 'rgb(255, 255, 13)';
                } else {
                    statusValueElement.style.color = 'red';
                    orderElement.style.borderLeftColor = 'red';
                }
            });
        }








        function goToCart() {
            window.location.href = 'Cart.html?mobile=' + mobile;
        }
        window.goToCart = goToCart;
        loadOrders();
